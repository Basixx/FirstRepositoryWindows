CREATE TABLE ISSUESLISTS
(
    ID SERIAL PRIMARY KEY,
    NAME VARCHAR (100)
);

CREATE TABLE ISSUES(
    ID SERIAL PRIMARY KEY,
    ISSUELIST_ID INT UNSIGNED NOT NULL,
    SUMMARY VARCHAR (100),
    DESCRIPTION VARCHAR (250),
    USER_ID_ASSIGNEDTO BIGINT UNSIGNED NOT NULL,
    FOREIGN KEY (USER_ID_ASSIGNEDTO) REFERENCES USERS(ID)
);
